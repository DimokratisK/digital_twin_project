#--------------
# PRE-TRAINING
#--------------

# 1. Clone the repo
git clone https://github.com/DimokratisK/digital_twin_project.git
cd digital_twin_project
git clone https://github.com/MIC-DKFZ/nnUNet.git nnUNet


# 2. Edit environment_nnunet.yml: change cu126 to match lab's GPU 

# 3. Create and activate the conda environment
conda env create -f environment_nnunet.yml
conda activate lab_venv_nnunet

# 4. Install nnU-Net from the cloned repo
pip install -e ./nnUNet

# 5. Copy the provided "nnunet_data" folder:
#    - Default: paste into <project_root>/nnunet_data
#    - Scratch disk: paste into /scratch/user/nnunet_data (or similar)

# 6. If data lives on a scratch disk (must be set in every new terminal session):
export NNUNET_DATA_ROOT=/scratch/user/nnunet_data



#----------
# TRAINING
#----------

python -m twin_core.nnunet_pipeline.run_training --profile full


#---------------
# POST-TRAINING
#---------------

python -m twin_core.nnunet_pipeline.run_find_best_config
